```bash
 > ferfereh help
ferfereh cleanup
 . cleanup ferfereh.
ferfereh exif get \
	[-] \
	<filename.jpg>
 . get exif.
ferfereh exif install
 . install exif.
ferfereh exif put \
	[~backup,dryrun,lat=<lat>,lon=<lon>,validate] \
	<filename.jpg>
 . put exif.
ferfereh publish coords \
	[~downloads]
 . publish ferfereh coords.
ferfereh publish 3d-files
 . publish ferfereh 3d-files.
```
